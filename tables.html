Java Script

	$(document).ready(function(){
		$('#data-table').DataTable({
			sDom: "frt<'dt-footer'ip>",
			pagingType: "simple",
			bSort: true,
			bSortClasses: true,
			sScrollY: "auto",
			sScrollX: "100%",
			'oLanguage':{
				'sSearch': '',
				'oPaginate': {
					'sNext': '',
					'sPrevious': ''
				}
			
			},
			'aoColumns': [
				{ "sClass": "cpl-title" },
				{ "sWidth": "160px" },
				{ "sWidth": "160px" },
				null,
				null,
				null
			]
		});
		$('div.dataTables_filter').appendTo('.table-filters'); //Places search filter outside table
		$('.main-content input[type="search"]').after(''); // Adds search icon to datatable search input
		$('.sort-table.dataTable th').append(''); //Adds span for arrow down for sorting table
	});
